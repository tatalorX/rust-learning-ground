Unit Tests

Unit tests verify individual components in isolation:

// src/lib.rs
pub fn add(a: i32, b: i32) -> i32 {
    a + b
}

#[cfg(test)]
mod tests {
    use super::*;
    
    #[test]
    fn test_add() {
        assert_eq!(add(2, 2), 4);
    }
    
    #[test]
    #[should_panic]
    fn test_overflow() {
        // This test expects a panic
        let _ = i32::MAX + 1;
    }
}

Integration Tests

Integration tests verify that components work together:

// tests/integration_test.rs
use my_crate::Database;

#[test]
fn test_database_connection() {
    let db = Database::connect("test.db").unwrap();
    db.insert("key", "value").unwrap();
    let value = db.get("key").unwrap();
    assert_eq!(value, "value");
}

Documentation Tests

Examples in documentation are tested to ensure they work:

Property-Based Testing

Instead of testing specific cases, test properties that should always hold:

use proptest::prelude::*;

proptest! {
    #[test]
    fn test_add_commutative(a: i32, b: i32) {
        // Addition should be commutative
        assert_eq!(add(a, b), add(b, a));
    }
}
