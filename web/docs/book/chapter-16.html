The Builder Pattern

Complex configuration becomes elegant with the builder pattern:

// Without builder
let config = Config {
    host: "localhost".to_string(),
    port: 8080,
    timeout: Duration::from_secs(30),
    retries: 3,
};

// With builder
let config = Config::builder()
    .host("localhost")
    .port(8080)
    .timeout(Duration::from_secs(30))
    .retries(3)
    .build()?;

Implementing Standard Traits

Make your types usable by implementing standard traits:

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct Point {
    x: i32,
    y: i32,
}

impl Default for Point {
    fn default() -> Self {
        Point { x: 0, y: 0 }
    }
}

impl fmt::Display for Point {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        write!(f, "({}, {})", self.x, self.y)
    }
}

Minimize your public API surface:

// Public API (stable, documented)
pub mod client;
pub use client::Client;

// Private implementation (can change)
mod internal;
mod protocol;
